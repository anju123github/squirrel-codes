Multithreading

2. Write a Java program that creates two threads to find and print even and odd numbers from 1 to 20.

Code


                

                class EvenOddPrinter extends Thread {
                    private final int start;
                    private final int end;
                
                    public EvenOddPrinter(int start, int end) {
                        this.start = start;
                        this.end = end;
                    }
                
                    @Override
                    public void run() {
                        if (start % 2 == 0) {
                            printEven();
                        } else {
                            printOdd();
                        }
                    }
                
                    private void printEven() {
                        for (int i = start; i <= end; i += 2) {
                            System.out.println(Thread.currentThread().getName() + ": " + i);
                        }
                    }
                
                    private void printOdd() {
                        for (int i = start; i <= end; i += 2) {
                            System.out.println(Thread.currentThread().getName() + ": " + i);
                        }
                    }
                }
                
                public class EvenOddThreadExample {
                    public static void main(String[] args) {
                        int start = 1;
                        int end = 20;
                
                        EvenOddPrinter evenPrinter = new EvenOddPrinter(start, end);
                        EvenOddPrinter oddPrinter = new EvenOddPrinter(start + 1, end);
                
                        Thread evenThread = new Thread(evenPrinter, "EvenThread");
                        Thread oddThread = new Thread(oddPrinter, "OddThread");
                
                        evenThread.start();
                        oddThread.start();
                    }
                }
